Data Preprocessing¶

In [0]:
# pick only 'Close' price and convert it to the nunmpy array
data = df.filter(['Close'])
# convert dataframe to np.array
dataset = data.values

# get 80% number of data as training size
# why not use random sampling? 
train_size = math.ceil(len(dataset)*0.8)
# create train dataset 
train_data = dataset[:train_size]
In [0]:
# scale the data
scaler = MinMaxScaler(feature_range=(0,1))

scaler = scaler.fit(train_data) # scaling should use the training dataset only
scaled_data = scaler.transform(dataset)
train_data = scaled_data[0:train_size,:]

In [0]:
# split traininng data into x and y
x_train = []
y_train = []

# use previous 60 days data as feature variables
# and the latest day price as the target variable
for i in range(60, train_size):
  x_train.append(train_data[i-60:i,0])
  y_train.append(train_data[i,0])
  
# convert x_train and y_train to np.array
x_train, y_train = np.array(x_train), np.array(y_train)

# reshape the data for LSTM model as a 3D numpy array
x_train = np.reshape(x_train, (x_train.shape[0], x_train.shape[1], 1))

Build the Model¶

In [8]:
# build a sequential model with two LSTM layers and two Dense layers
model = Sequential()
model.add(LSTM(50, return_sequences=True, input_shape = (x_train.shape[1], 1)))
model.add(LSTM(50, return_sequences=False))
model.add(Dense(25))
model.add(Dense(1))

# compile the model
model.compile(optimizer='adam', loss='mean_squared_error')

# train the model
model.fit(x_train, y_train, batch_size=1, epochs=1)
